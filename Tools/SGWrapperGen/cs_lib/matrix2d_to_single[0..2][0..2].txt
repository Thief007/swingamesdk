
[System.Diagnostics.DebuggerNonUserCode(), System.Diagnostics.DebuggerStepThrough()]
public static implicit operator float[,](Matrix2D m)
{
    return m.data;
}

[System.Diagnostics.DebuggerNonUserCode(), System.Diagnostics.DebuggerStepThrough()]
public static implicit operator Matrix2D(float[,] data)
{
    Matrix2D result;
    result.data = data;
    return result;
}

function %(name)s(%(params)s): %(return_type)s; cdecl; export;
begin
  {$IFDEF TRACE}
    TraceEnter('SGSDK.dll', '%(name)s');
  {$ENDIF}
  
  try
    result := %(calls.file.pascal_name)s.%(calls.name)s(%(calls.args)s);
  Except on exc: Exception do 
    TrapException(exc, '%(name)s');
  end;  
  
  {$IFDEF TRACE}
    TraceExit('SGSDK.dll', '%(name)s');
  {$ENDIF}
end;
